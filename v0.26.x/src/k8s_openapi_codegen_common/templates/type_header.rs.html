1pub(crate) fn generate(
2    mut writer: impl std::io::Write,
3    definition_path: &crate::swagger20::DefinitionPath,
4    type_comment: Option<&str>,
5    derives: Option<Derives>,
6    vis: &str,
7) -> Result<(), crate::Error> {
8    let type_comment =
9        type_comment
10        .map(|type_comment| crate::get_comment_text(type_comment, "").fold(String::new(), |mut result, line| {
11            result.push_str("///");
12            result.push_str(&line);
13            result.push('\n');
14            result
15        }))
16        .unwrap_or_default();
17
18    let derives =
19        derives
20        .map(|Derives { clone, copy, default, eq, ord, partial_eq, partial_ord }| format!(
21            "#[derive({clone}{copy}Debug{default}{eq}{ord}{partial_eq}{partial_ord})]\n",
22            clone = if clone { "Clone, " } else { "" },
23            copy = if copy { "Copy, " } else { "" },
24            default = if default { ", Default" } else { "" },
25            eq = if eq { ", Eq" } else { "" },
26            ord = if ord { ", Ord" } else { "" },
27            partial_eq = if partial_eq { ", PartialEq" } else { "" },
28            partial_ord = if partial_ord { ", PartialOrd" } else { "" },
29        ))
30        .unwrap_or_default();
31
32    write!(
33        writer,
34        include_str!(concat!(env!("CARGO_MANIFEST_DIR"), "/templates/type_header.rs")),
35        definition_path = definition_path,
36        type_comment = type_comment,
37        derives = derives,
38        vis = vis,
39    )?;
40
41    Ok(())
42}
